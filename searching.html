1. Linear Search

Linear Search adalah algoritma pencarian yang paling sederhana. Algoritma ini memeriksa setiap elemen array satu per satu dari indeks pertama hingga terakhir, hingga elemen yang dicari (target) ditemukan atau seluruh array telah diperiksa. Linear Search dapat digunakan pada array yang tidak terurut maupun yang sudah terurut.

Kode Program (Bagian: linearSearch)

C++ 
int linearSearch(int arr[], int size, int target)
{
    for (int i = 0; i < size; i++)
    {
        if (arr[i] == target) // Periksa setiap elemen
        {
            return i;         // Kembalikan indeks jika ditemukan
        }
    }
    return -1;                // Kembalikan -1 jika tidak ditemukan
}

#include <iostream>
using namespace std;

const int SIZE = 10;

int linearSearch(int arr[], int size, int target)
{
    for (int i = 0; i < size; i++)
    {
        if (arr[i] == target)
        {
            return i;
        }
    }
    return -1;
}

int main()
{
    int arr[SIZE] = {10, 20, 30, 40, 50, 60, 70, 80, 90, 91};
    int target;

    cout << "Masukkan angka yang ingin dicari: ";
    cin >> target;

    cout << "\nMelakukan Linear Search...." << endl;
    int index = linearSearch(arr, SIZE, target);

    if (index != -1)
    {
        cout << "\nAngka " << target << " ditemukan pada index ke-" << index << "." << endl;
    }
    else
    {
        cout << "\nAngka " << target << " tidak ditemukan dalam array." << endl;
    }

    return 0;
}

Penjelasan Program

Cara Kerja Linear Search:
  • Mulai dari indeks pertama (i = 0).
  • Bandingkan setiap elemen arr[i] dengan nilai target.
  • Jika cocok, kembalikan indeks i sebagai posisi elemen ditemukan.
  • Jika seluruh array telah diperiksa tanpa kecocokan, kembalikan -1.
  • Array tidak perlu diurutkan terlebih dahulu.
⏱ Kompleksitas:
  • Best Case: O(1) — elemen ditemukan di indeks pertama.
  • Average & Worst Case: O(n) — elemen di akhir atau tidak ada.
  • Space Complexity: O(1) — tidak memerlukan memori tambahan.

2. Binary Search

Binary Search adalah algoritma pencarian yang jauh lebih efisien dibandingkan Linear Search, namun hanya dapat digunakan pada array yang sudah terurut. Algoritma ini bekerja dengan membagi area pencarian menjadi dua secara berulang, membandingkan nilai tengah (mid) dengan target, lalu memilih bagian kiri atau kanan untuk dilanjutkan.

Pada contoh berikut, array diurutkan terlebih dahulu menggunakan Selection Sort sebelum Binary Search dijalankan.

Kode Program (Bagian: selectionSort)

C++ 
void selectionSort(int arr[], int size)
{
    for (int i = 0; i < size - 1; i++)
    {
        int minIndex = i;
        for (int j = i + 1; j < size; j++)
        {
            if (arr[j] < arr[minIndex]) // Cari elemen terkecil
            {
                minIndex = j;
            }
        }
        swap(arr[i], arr[minIndex]); // Tukar ke posisi yang benar
    }
}

Kode Program (Bagian: binarySearch)

C++ 
int binarySearch(int arr[], int size, int target)
{
    int low  = 0;
    int high = size - 1;

    while (low <= high)
    {
        int mid = low + (high - low) / 2; // Hitung indeks tengah

        if (arr[mid] == target)
            return mid;           // Ditemukan

        if (arr[mid] < target)
            low = mid + 1;        // Cari di bagian kanan
        else
            high = mid - 1;       // Cari di bagian kiri
    }
    return -1;                    // Tidak ditemukan
}

C++ — Full Code 
#include <iostream>
using namespace std;

const int SIZE = 10;

void selectionSort(int arr[], int size)
{
    for (int i = 0; i < size - 1; i++)
    {
        int minIndex = i;
        for (int j = i + 1; j < size; j++)
        {
            if (arr[j] < arr[minIndex])
            {
                minIndex = j;
            }
        }
        swap(arr[i], arr[minIndex]);
    }
}

int binarySearch(int arr[], int size, int target)
{
    int low  = 0;
    int high = size - 1;

    while (low <= high)
    {
        int mid = low + (high - low) / 2;

        if (arr[mid] == target)
            return mid;

        if (arr[mid] < target)
            low = mid + 1;
        else
            high = mid - 1;
    }
    return -1;
}

void displayArray(int arr[], int size)
{
    cout << "[ ";
    for (int i = 0; i < size; i++)
        cout << arr[i] << " ";
    cout << "]" << endl;
}

int main()
{
    int arr[SIZE] = {95, 75, 55, 15, 5, 45, 35, 65, 25, 85};

    selectionSort(arr, SIZE);

    cout << "Array setelah diurutkan: ";
    displayArray(arr, SIZE);

    int target;
    cout << "Masukkan angka yang ingin dicari: ";
    cin >> target;

    cout << "\nMelakukan Binary Search..." << endl;
    int index = binarySearch(arr, SIZE, target);

    if (index != -1)
    {
        cout << "\nAngka " << target << " ditemukan pada index ke-" << index << " (setelah diurutkan)." << endl;
    }
    else
    {
        cout << "\nAngka " << target << " tidak ditemukan dalam array." << endl;
    }

    return 0;
}

Penjelasan Program

Cara Kerja Binary Search:
  • Array harus diurutkan terlebih dahulu (di sini menggunakan Selection Sort).
  • Tentukan pointer low = 0 dan high = size - 1.
  • Hitung mid = low + (high - low) / 2.
  • Jika arr[mid] == target → elemen ditemukan, kembalikan mid.
  • Jika arr[mid] < target → geser low = mid + 1 (cari kanan).
  • Jika arr[mid] > target → geser high = mid - 1 (cari kiri).
  • Ulangi hingga low > high. Jika tidak ditemukan, kembalikan -1.
⏱ Kompleksitas:
  • Best Case: O(1) — target ada tepat di tengah.
  • Average & Worst Case: O(log n) — area pencarian dibagi dua setiap iterasi.
  • Space Complexity: O(1) — tidak memerlukan memori tambahan.
  • Syarat: Array harus terurut sebelum Binary Search dijalankan.
