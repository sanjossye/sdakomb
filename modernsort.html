1. Quick Sort

Quick Sort adalah algoritma Divide and Conquer yang sangat efisien. Algoritma ini memilih sebuah elemen sebagai pivot dan mempartisi array di sekitar pivot tersebut sehingga elemen yang lebih kecil berada di kiri dan yang lebih besar berada di kanan.

Kode Program (Bagian: partition)

C++ 
int partition(int arr[], int low, int high)
{
    int pivot = arr[high]; // Pilih pivot (elemen terakhir)
    int i = low - 1;       // Indeks elemen yang lebih kecil

    for (int j = low; j < high; j++)
    {
        if (arr[j] <= pivot) // Jika elemen lebih kecil dari pivot
        {
            i++;
            swap(arr[i], arr[j]); // Tukar elemen agar berada di kiri pivot
        }
    }

    i++;
    swap(arr[i], arr[high]); // Tempatkan pivot di posisi yang benar
    return i;                // Kembalikan posisi pivot
}

Kode Program (Bagian: quickSort)

C++ 
void quickSort(int arr[], int low, int high)
{
    if (high <= low)
        return;

    int pivotIndex = partition(arr, low, high); // Partisi array

    quickSort(arr, low, pivotIndex - 1);  // Urutkan bagian kiri
    quickSort(arr, pivotIndex + 1, high); // Urutkan bagian kanan
}

#include <iostream>
using namespace std;

int partition(int arr[], int low, int high)
{
    int pivot = arr[high];
    int i = low - 1;

    for (int j = low; j < high; j++)
    {
        if (arr[j] <= pivot)
        {
            i++;
            swap(arr[i], arr[j]);
        }
    }

    i++;
    swap(arr[i], arr[high]);
    return i;
}

void quickSort(int arr[], int low, int high)
{
    if (high <= low)
        return;

    int pivotIndex = partition(arr, low, high);

    quickSort(arr, low, pivotIndex - 1);
    quickSort(arr, pivotIndex + 1, high);
}

void displayArray(int arr[], int length)
{
    cout << "[ ";
    for (int i = 0; i < length; i++)
        cout << arr[i] << " ";
    cout << "]" << endl;
}

int main()
{
    int arr[] = {70, 10, 80, 60, 90};
    int length = sizeof(arr) / sizeof(arr[0]);

    cout << "Array sebelum sorting: ";
    displayArray(arr, length);

    quickSort(arr, 0, length - 1);

    cout << "Array setelah sorting: ";
    displayArray(arr, length);

    return 0;
}

Penjelasan Program

Cara Kerja Quick Sort:
  • Pilih elemen pivot (dalam contoh ini elemen terakhir).
  • Partitioning: Geser semua elemen yang lebih kecil dari pivot ke sebelah kirinya, dan elemen yang lebih besar ke kanannya.
  • Kembalikan indeks pivot yang sudah berada di posisi final.
  • Rekursi panggil quickSort untuk bagian kiri dan kanan pivot.
⏱ Kompleksitas:
  • Best & Average Case: O(n log n)
  • Worst Case: O(n²) (terjadi jika pivot selalu elemen terkecil/terbesar)
  • Space Complexity: O(log n) untuk rekursi stack.

2. Merge Sort

Merge Sort juga menggunakan prinsip Divide and Conquer. Algoritma ini terus membagi array menjadi dua bagian hingga masing-masing bagian hanya berisi satu elemen, kemudian menggabungkannya kembali (merge) dalam urutan yang benar.

Kode Program (Bagian: merge)

C++ 
void merge(int arr[], int left, int mid, int right)
{
    int n1 = mid - left + 1;
    int n2 = right - mid;

    int leftArr[n1], rightArr[n2];

    for (int i = 0; i < n1; i++)
        leftArr[i] = arr[left + i];
    for (int j = 0; j < n2; j++)
        rightArr[j] = arr[mid + 1 + j];

    int i = 0, j = 0, k = left;
    while (i < n1 && j < n2)
    {
        if (leftArr[i] <= rightArr[j])
            arr[k++] = leftArr[i++];
        else
            arr[k++] = rightArr[j++];
    }

    while (i < n1) arr[k++] = leftArr[i++];
    while (j < n2) arr[k++] = rightArr[j++];
}

Kode Program (Bagian: mergeSort)

C++ 
void mergeSort(int arr[], int left, int right)
{
    if (left >= right)
        return;

    int mid = left + (right - left) / 2;

    mergeSort(arr, left, mid);
    mergeSort(arr, mid + 1, right);

    merge(arr, left, mid, right);
}

#include <iostream>
using namespace std;

void merge(int arr[], int left, int mid, int right)
{
    int n1 = mid - left + 1;
    int n2 = right - mid;

    int leftArr[n1], rightArr[n2];

    for (int i = 0; i < n1; i++)
        leftArr[i] = arr[left + i];
    for (int j = 0; j < n2; j++)
        rightArr[j] = arr[mid + 1 + j];

    int i = 0, j = 0, k = left;
    while (i < n1 && j < n2)
    {
        if (leftArr[i] <= rightArr[j])
            arr[k++] = leftArr[i++];
        else
            arr[k++] = rightArr[j++];
    }
    while (i < n1) arr[k++] = leftArr[i++];
    while (j < n2) arr[k++] = rightArr[j++];
}

void mergeSort(int arr[], int left, int right)
{
    if (left >= right)
        return;

    int mid = left + (right - left) / 2;

    mergeSort(arr, left, mid);
    mergeSort(arr, mid + 1, right);

    merge(arr, left, mid, right);
}

void displayArray(int arr[], int length)
{
    cout << "[ ";
    for (int i = 0; i < length; i++)
        cout << arr[i] << " ";
    cout << "]" << endl;
}

int main()
{
    int arr[] = {80, 20, 70, 90};
    int length = sizeof(arr) / sizeof(arr[0]);

    cout << "Array sebelum sorting: ";
    displayArray(arr, length);

    mergeSort(arr, 0, length - 1);

    cout << "Array setelah sorting: ";
    displayArray(arr, length);

    return 0;
}

Penjelasan Program

Cara Kerja Merge Sort:
  • Bagi array menjadi dua bagian (kiri dan kanan).
  • Lakukan pembagian secara rekursif hingga ukuran subarray adalah 1.
  • Merge: Bandingkan elemen dari kedua subarray dan susun kembali ke array utama secara terurut.
⏱ Kompleksitas:
  • All Cases: O(n log n) — sangat stabil kinerjanya.
  • Space Complexity: O(n) karena membutuhkan array sementara untuk proses merging.

Contoh Output

Array sebelum sorting: [ 80 20 70 90 ]
Array setelah sorting: [ 20 70 80 90 ]
